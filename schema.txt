model User {
  id        Int      @id @default(autoincrement())
  email     String   @unique
  name      String
  password  String 
  profilePic Int
}

model Post {
    id Int @id @default(autoincrement())
    title String
    content String
    date DateTime @default(now())
    likes Int
    userid Int
}

model Image {
    id Int @id @default(autoincrement())
    url String
    likes Int
    forProfile Boolean
    forId Int
}

model Comment {
    id Int @id @default(autoincrement())
    date DateTime @default(now())
    content String
    authorid Int
    forImg boolean
    forId Int
}

model Friendships {
    id Int @id @default(autoincrement())
    userOne Int
    userTwo Int
    status String
}

SELECT u.id, u.username, u.avatar_url
FROM users u
JOIN (
    SELECT user_id2 AS friend_id FROM friendships WHERE user_id1 = 3 AND status = 'accepted'
    UNION ALL
    SELECT user_id1 AS friend_id FROM friendships WHERE user_id2 = 3 AND status = 'accepted'
) f ON u.id = f.friend_id;
