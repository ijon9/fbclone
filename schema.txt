model User {
  id        Int      @id @default(autoincrement())
  email     String   @unique
  name      String
  password  String 
  profilePic Int
}

model Post {
    id Int @id @default(autoincrement())
    date DateTime @default(now())
    title String
    content String
    userid Int
}

model Image {
    id Int @id @default(autoincrement())
    date DateTime @default(now())
    url String
    profileId Int
    postId Int
}

model Comment {
    id Int @id @default(autoincrement())
    date DateTime @default(now())
    content String
    authorId Int
    imgId boolean
    postId Int
}

model Friendships {
    id Int @id @default(autoincrement())
    userOne Int
    userTwo Int
    status String
}

model Likes {
    userId Int
    postId Int
    imgId Int
}

SELECT f1.friend_id AS mutual_friend_id
FROM friendships f1
INNER JOIN friendships f2 
   ON f1.friend_id = f2.friend_id
WHERE f1.user_id = :user_a_id 
  AND f2.user_id = :user_b_id;
